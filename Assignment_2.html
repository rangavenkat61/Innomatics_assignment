Building Polynimal Regression¶

In [28]:
score_dict = {}
for i in range(1,8):
    pr = PolynomialFeatures(degree=i,include_bias = False)
    x_train_transformed = pr.fit_transform(x_train.values.reshape(-1,1))
    x_test_transformed = pr.transform(x_test.values.reshape(-1,1))
    lr = LinearRegression()
    lr.fit(x_train_transformed,y_train)
    score_dict[i] = lr.score(x_test_transformed,y_test)
